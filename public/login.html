async function registrarse() { const btn = document.querySelector('button'); btn.disabled = true; // Bloqueamos el botón para evitar duplicados btn.innerText = "Cargando..."; const data = { username: document.getElementById('reg_user').value, email: document.getElementById('reg_email').value, password: document.getElementById('reg_pass').value }; try { const response = await fetch('/api/auth/register', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(data) }); const result = await response.json(); if (result.success) { alert("¡ÉXITO! Ahora inicia sesión."); location.reload(); } else { alert("ERROR: " + result.error); } } catch (error) { alert("Error de conexión"); } finally { btn.disabled = false; btn.innerText = "CREAR CUENTA"; } }
